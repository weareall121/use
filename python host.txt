// 1. Environment Setup: Load environment variables from .env file
require('dotenv').config();
const cloudinary = require('cloudinary').v2;
const fs = require('fs');

// 2. Security: Configure Cloudinary with credentials from environment variables
cloudinary.config({
  cloud_name: process.env.CLOUDINARY_CLOUD_NAME,
  api_key: process.env.CLOUDINARY_API_KEY,
  api_secret: process.env.CLOUDINARY_API_SECRET,
  secure: true,
});

// 3. Image Fetching and Link Generation
async function generateImageLinksFile() {
  console.log('Fetching images from Cloudinary...');
  try {
    const searchResult = await cloudinary.search
      .expression('') // Get all images
      .sort_by('public_id', 'desc')
      .max_results(500)
      .execute();

    console.log(`${searchResult.resources.length} images found.`);

    // 4. Optimization: Generate optimized URLs
    const imageUrls = searchResult.resources.map(image => {
      const optimizedUrl = cloudinary.url(image.public_id, {
        fetch_format: 'auto',
        quality: 'auto',
        secure: true
      });
      return optimizedUrl;
    }).join('\n');

    // 5. File Generation: Write the URLs to a text file
    fs.writeFileSync('cloudinary_links.txt', imageUrls);
    console.log('Successfully created cloudinary_links.txt!');

  } catch (error) {
    console.error('Error fetching images or creating file:', error);
  }
}

generateImageLinksFile();
EOF
